        /* 画面別個別機能ココマデ */
        //-------------------------------------------------------------------------------------------------------------------------------------
        // 件数が無い場合は NOTFOUND_ERROR
        //-------------------------------------------------------------------------------------------------------------------------------------
        if return_strings == "\"\"" {
            return BusinessLogic {
                result:5,
                data:seg4_common::NOTFOUND_ERROR.to_string(), 
            }
        };

        //---------------------------------------------------------------------------------------------------------------------------------
        // 戻り値として返却:※正常終了
        //---------------------------------------------------------------------------------------------------------------------------------
        BusinessLogic {
            result:0,
            //data:format!("{{\"result\":{}}}",return_strings),
            data:format!("{{ \"result\":\"200 reqest complete\",\"api_data\": {} }}",return_strings),
        }
    } //execute 終端
}//impl 終端
